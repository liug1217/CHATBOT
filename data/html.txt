<!-- 範例:表格資料依欄位排序 -->
<table id="dataTable" border="1">
    <thead>
        <tr>
            <th onclick="sortTable(0)">姓名</th>
            <th onclick="sortTable(1)">分數</th>
        </tr>
    </thead>
    <tbody>
        <tr><td>小明</td><td>85</td></tr>
        <tr><td>小華</td><td>92</td></tr>
        <tr><td>小美</td><td>78</td></tr>
    </tbody>
</table>

<script>
function sortTable(colIndex) {
    const table = document.getElementById("dataTable");
    const rows = Array.from(table.tBodies[0].rows);
    rows.sort((a, b) => a.cells[colIndex].textContent.localeCompare(b.cells[colIndex].textContent));
    rows.forEach(row => table.tBodies[0].appendChild(row));
}
</script>

<!-- 範例:簡單的下拉搜尋自動完成 -->
<input type="text" id="searchInput" placeholder="搜尋水果..." oninput="showSuggestions()">
<ul id="suggestions"></ul>

<script>
const fruits = ["蘋果", "香蕉", "橘子", "芭樂", "西瓜", "葡萄"];

function showSuggestions() {
    const input = document.getElementById("searchInput").value;
    const list = document.getElementById("suggestions");
    list.innerHTML = "";
    if (!input) return;
    const matches = fruits.filter(f => f.includes(input));
    matches.forEach(match => {
        const li = document.createElement("li");
        li.textContent = match;
        list.appendChild(li);
    });
}
</script>

<script>
document.getElementById("tagInput").addEventListener("keydown", function(e) {
    if (e.key === "Enter" && this.value.trim() !== "") {
        const tag = document.createElement("span");
        tag.textContent = this.value + " ";
        tag.style.background = "#e0e0e0";
        tag.style.padding = "2px 8px";
        tag.style.borderRadius = "4px";
        tag.style.marginRight = "4px";
        document.getElementById("tagContainer").appendChild(tag);
        this.value = "";
    }
});
</script>

<!-- 範例:簡單的倒數計時器 -->
<div id="countdown">10</div>

<script>
let count = 10;
const timer = setInterval(() => {
    count--;
    document.getElementById("countdown").textContent = count;
    if (count <= 0) {
        clearInterval(timer);
        document.getElementById("countdown").textContent = "時間到!";
    }
}, 1000);
</script>

    <label><input type="checkbox" onchange="updateCount()"> 選項三</label><br>
</div>
<p>已選擇: <span id="selectedCount">0</span> 項</p>

<script>
function updateCount() {
    const checked = document.querySelectorAll("#checkboxList input:checked").length;
    document.getElementById("selectedCount").textContent = checked;
}
</script>
